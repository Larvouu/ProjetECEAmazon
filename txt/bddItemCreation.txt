create database if not exists eceamazon;
use eceamazon;
drop table if exists item;


CREATE TABLE item(
    id int(250) NOT NULL AUTO_INCREMENT, 
    nom varchar(100), 
    photo varchar(200),
    description varchar(200),
    categorie varchar(200),
    prix float,
    video varchar(200),
    isVendu boolean DEFAULT '0',
    isPanier boolean DEFAULT '0',
    tailleVetement varchar(10) DEFAULT '',
    tailleChaussure varchar(5) DEFAULT '',
    couleur varchar(200) DEFAULT '',
    auteur varchar(200) DEFAULT '',
    vendeur_email varchar(100),
    PRIMARY KEY (id),
    FOREIGN KEY (vendeur_email) REFERENCES vendeur (email)
);
