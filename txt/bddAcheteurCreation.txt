create database if not exists eceamazon;
use eceamazon;
drop table if exists acheteur;


CREATE TABLE acheteur(
    email varchar(100), 
    mdp varchar(100),
    nom varchar(100),
    prenom varchar(100),
    ville varchar(100),
    adresse varchar(100),
    codePostal int(100),
    pays varchar(100),
    numTel int(100),
    numCarte int(100),
    typeCarte varchar(100),
    nomAfficheCarte varchar(100),
    dateExpi varchar(100),
    codeSecu int(100),
    PRIMARY KEY (email));
